**Goal:** Implement the five LangGraph node functions (consolidator, planner, executor, evaluator, summarizer) in `src/clinical_codes/graph/nodes.py` with full offline test coverage.

**Architecture:** All five nodes in one file; three module-level `ChatAnthropic` chain singletons (planner, evaluator, summarizer) constructed at import time; consolidator is a pure sync function; executor fans out with `asyncio.gather` and merges results back into `raw_results`; LLM nodes are tested by monkeypatching module-level chains; consolidator tests need no mocks.

**Tech Stack:** `langchain-anthropic` (`ChatAnthropic`, `.with_structured_output()`), `asyncio`, `pytest-asyncio` (`asyncio_mode = "auto"` already set), `unittest.mock` (`AsyncMock`, `MagicMock`, `patch`).

from clinical_codes.config import settings
from clinical_codes.graph.state import GraphState
from clinical_codes.schemas import CodeResult, SystemName


def consolidator(state: GraphState) -> dict:
    selected = state["planner_output"].selected_systems
    raw = state["raw_results"]

    consolidated: dict[SystemName, list[CodeResult]] = {}
    for system in selected:
        results = raw.get(system, [])
        seen: set[str] = set()
        deduped: list[CodeResult] = []
        for r in results:
            if r.code not in seen:
                seen.add(r.code)
                deduped.append(r)
        deduped.sort(key=lambda r: r.score, reverse=True)
        consolidated[system] = deduped[:settings.display_results]

    return {"consolidated": consolidated}
```

async def executor(state: GraphState) -> dict:
    search_terms = state["planner_output"].search_terms

    async def _search_one(
        system: SystemName, term: str
    ) -> tuple[SystemName, list[CodeResult]]:
        async with CLIENTS[system]() as client:
            results = await client.search(term)
        return system, results

    pairs: list[tuple[SystemName, list[CodeResult]]] = await asyncio.gather(
        *[_search_one(system, term) for system, term in search_terms.items()]
    )
    merged = dict(state["raw_results"])
    for system, results in pairs:
        merged[system] = results
    return {"raw_results": merged}
```
